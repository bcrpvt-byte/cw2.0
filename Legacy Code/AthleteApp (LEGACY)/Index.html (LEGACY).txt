        // Helper function to safely parse arrays (handling double-stringification from GAS)
        const safeArrayParse = (data) => {
            if (!data) return null;
            if (Array.isArray(data)) return data;
            if (typeof data === 'string') {
                try {
                    const parsed = JSON.parse(data);
                    if (Array.isArray(parsed)) return parsed;
                    if (typeof parsed === 'string') {
                        const doubleParsed = JSON.parse(parsed);
                        if (Array.isArray(doubleParsed)) return doubleParsed;
                    }
                } catch (e) {
                    console.error("Failed to parse array data:", data, e);
                    return null;
                }
            }
            return null;
        };

        // Custom Component für Icons, da lucide-react in diesem Setup nicht direkt funktioniert
        // Wir nutzen das globale "lucide" Objekt, um SVG Strings zu generieren und rendern diese via dangerouslySetInnerHTML
        const Icon = ({ name, size = 24, className = '', strokeWidth = 2 }) => {
            // Lucide Icons global keys are PascalCase (e.g. CheckCircle2)
            // But sometimes might be requested by the component differently. We just map straight to the name.
            let iconName = name;
            
            // Allow matching names properly with or without case difference
            // Also replace dashes which are often requested in standard lucide React usage
            const normalizedRequestedName = name.replace(/-/g, '').toLowerCase();

            if (!lucide.icons[iconName]) {
                 const match = Object.keys(lucide.icons).find(k => k.toLowerCase() === normalizedRequestedName);
                 if (match) {
                     iconName = match;
                 }
            }

            // Fallback, falls Icon nicht gefunden
            if (!lucide.icons[iconName]) {
                 console.warn(`Icon ${name} not found in lucide.`);
                 return <span className={className} style={{ width: size, height: size, display: 'inline-block' }}></span>;
            }

            // Build the SVG using React elements since toSvg() is deprecated/removed in some unpkg bundles
            const iconData = lucide.icons[iconName];
            
            const children = iconData.map(([tag, attrs], index) => {
                return React.createElement(tag, { ...attrs, key: index });
            });
            
            return React.createElement(
                'svg',
                {
                    width: size,
                    height: size,
                    viewBox: "0 0 24 24",
                    fill: "none",
                    stroke: "currentColor",
                    strokeWidth: strokeWidth,
                    strokeLinecap: "round",
                    strokeLinejoin: "round",
                    className: `lucide lucide-${iconName} ${className}`
                },
                children
            );
        };

        function ScannerInitializer({ onScanSuccess, onScanFailure, webAppUrl }) {
            const html5QrcodeScannerRef = React.useRef(null);
            const [errorMsg, setErrorMsg] = React.useState('');
            const [useFallback, setUseFallback] = React.useState(false);
            const [isProcessingFile, setIsProcessingFile] = React.useState(false);

            const processFile = async (e) => {
                const file = e.target.files[0];
                if (!file) return;

                setIsProcessingFile(true);
                setErrorMsg('');

                try {
                    // 1. Load image to scale down
                    const img = new Image();
                    const imageUrl = URL.createObjectURL(file);

                    await new Promise((resolve, reject) => {
                        img.onload = resolve;
                        img.onerror = reject;
                        img.src = imageUrl;
                    });

                    // 2. Downscale image
                    const canvas = document.createElement('canvas');
                    const MAX_DIMENSION = 800; // 800px is usually enough for barcodes and fast to process
                    let width = img.width;
                    let height = img.height;

                    if (width > MAX_DIMENSION || height > MAX_DIMENSION) {
                        if (width > height) {
                            height = Math.round((height * MAX_DIMENSION) / width);
                            width = MAX_DIMENSION;
                        } else {
                            width = Math.round((width * MAX_DIMENSION) / height);
                            height = MAX_DIMENSION;
                        }
                    }
                    canvas.width = width;
                    canvas.height = height;
                    const ctx = canvas.getContext('2d');
                    ctx.drawImage(img, 0, 0, width, height);

                    let foundCode = null;

                    // 3. Try native BarcodeDetector if available (Very fast, native API)
                    if ('BarcodeDetector' in window) {
                        try {
                            const barcodeDetector = new window.BarcodeDetector({
                                formats: ['ean_13', 'ean_8', 'upc_a', 'upc_e', 'qr_code', 'code_128', 'code_39']
                            });
                            const barcodes = await barcodeDetector.detect(canvas);
                            if (barcodes && barcodes.length > 0) {
                                foundCode = barcodes[0].rawValue;
                                console.log("Barcode via native BarcodeDetector gefunden:", foundCode);
                            }
                        } catch (err) {
                            console.warn("BarcodeDetector error:", err);
                        }
                    }

                    if (foundCode) {
                        onScanSuccess(foundCode, { text: foundCode });
                    } else {
                        // 4. Fallback to html5-qrcode file scan
                        const blob = await new Promise(resolve => canvas.toBlob(resolve, 'image/jpeg', 0.8));
                        const scaledFile = new File([blob], "scaled_barcode.jpg", { type: "image/jpeg" });

                        const tempScanner = new Html5Qrcode("hidden-reader");
                        try {
                            const decodedText = await tempScanner.scanFile(scaledFile, false);
                            if (decodedText) {
                                console.log("Barcode via Html5Qrcode Fallback gefunden:", decodedText);
                                onScanSuccess(decodedText, { text: decodedText });
                            }
                        } catch (scanErr) {
                            console.error("Html5Qrcode file scan error:", scanErr);
                            setErrorMsg("Kein Barcode gefunden, bitte versuche es noch einmal oder trage die Nummer manuell ein.");
                        }
                    }
                } catch (err) {
                    console.error("File processing error:", err);
                    setErrorMsg("Fehler beim Verarbeiten des Bildes.");
                } finally {
                    setIsProcessingFile(false);
                }
            };

            React.useEffect(() => {
                let isMounted = true;

                const initScanner = async () => {
                    try {
                        // Request camera permissions first
                        const cameras = await Html5Qrcode.getCameras();

                        if (cameras && cameras.length > 0) {
                            if (!html5QrcodeScannerRef.current) {
                                html5QrcodeScannerRef.current = new Html5Qrcode("reader");
                            }

                            const config = { fps: 10, qrbox: { width: 250, height: 250 } };

                            await html5QrcodeScannerRef.current.start(
                                { facingMode: "environment" },
                                config,
                                (decodedText, decodedResult) => {
                                    // Stop scanning on success
                                    if (html5QrcodeScannerRef.current) {
                                        try {
                                            html5QrcodeScannerRef.current.stop().then(() => {
                                                if (isMounted) onScanSuccess(decodedText, decodedResult);
                                            }).catch(err => {
                                                console.error("Error stopping scanner", err);
                                                if (isMounted) onScanSuccess(decodedText, decodedResult);
                                            });
                                        } catch (e) {
                                            if (isMounted) onScanSuccess(decodedText, decodedResult);
                                        }
                                    } else {
                                        if (isMounted) onScanSuccess(decodedText, decodedResult);
                                    }
                                },
                                (errorMessage) => {
                                    if (onScanFailure) onScanFailure(errorMessage);
                                }
                            );
                        } else {
                            if (isMounted) {
                                setErrorMsg("Keine Kameras gefunden.");
                                setUseFallback(true);
                            }
                        }
                    } catch (err) {
                        console.error("Failed to start scanner:", err);
                        if (isMounted) {
                            setErrorMsg("Live-Kamera blockiert (Google Apps Script Iframe). Bitte App im Vollbild öffnen.");
                            setUseFallback(true);
                        }
                    }
                };

                initScanner();

                return () => {
                    isMounted = false;
                    if (html5QrcodeScannerRef.current) {
                        try {
                            if (html5QrcodeScannerRef.current.isScanning) {
                                html5QrcodeScannerRef.current.stop().catch(console.error);
                            }
                        } catch (e) {
                            console.error("Cleanup error:", e);
                        }
                    }
                };
            }, [onScanSuccess, onScanFailure]);

            return (
                <div style={{ width: "100%", maxWidth: "600px", margin: "0 auto" }}>
                    {errorMsg && (
                        <div className="p-4 mb-4 text-sm text-yellow-800 rounded-lg bg-yellow-50 dark:bg-yellow-900/30 dark:text-yellow-400" role="alert">
                            {errorMsg}
                        </div>
                    )}

                    <div id="reader" style={{ width: "100%", borderRadius: "12px", overflow: "hidden", display: useFallback ? 'none' : 'block' }}></div>
                    <div id="hidden-reader" style={{ display: "none" }}></div>

                    {useFallback && (
                        <div className="flex flex-col items-center justify-center p-6 bg-slate-100 dark:bg-slate-800 rounded-2xl border-2 border-dashed border-slate-300 dark:border-slate-600">
                            <Icon name="Camera" size={48} className="text-slate-400 mb-4" />
                            <p className="text-center text-slate-600 dark:text-slate-300 mb-4 font-medium">
                                Live-Kamera nicht verfügbar. Bitte mache ein Foto vom Barcode.
                            </p>

                            <label className="px-6 py-3 bg-blue-600 text-white rounded-full font-bold shadow-md hover:bg-blue-700 transition flex items-center justify-center gap-2 w-full text-center cursor-pointer">
                                {isProcessingFile ? (
                                    <>
                                        <div className="animate-spin h-5 w-5 border-2 border-white border-t-transparent rounded-full"></div>
                                        <span>Verarbeite Foto...</span>
                                    </>
                                ) : (
                                    <>
                                        <Icon name="Upload" size={20} />
                                        <span>Foto aufnehmen / auswählen</span>
                                    </>
                                )}
                                <input
                                    type="file"
                                    accept="image/*"
                                    capture="environment"
                                    className="hidden"
                                    onChange={processFile}
                                    disabled={isProcessingFile}
                                />
                            </label>

                            <p className="text-[10px] text-slate-500 text-center max-w-[250px] mt-4 pt-3 border-t border-slate-200 dark:border-slate-700">
                                Unser System verkleinert das Bild automatisch, damit der Barcode blitzschnell und zuverlässig erkannt wird.
                            </p>
                        </div>
                    )}
                </div>
            );
        }

                  }

                  estimatedDurationMinutes += Math.round((setsTime + restTime) / 60);

                  return {
                      ...ex,
                      sets: totalSets, // Normalize sets
                  };
              });

                        <span className="text-[10px] mt-1.5 font-bold">Plan</span>
                      </button>
                    </div>

                    {/* Quick-Action Button */}
                    <div className="absolute left-1/2 -translate-x-1/2 -top-10">
                      <button
                        onClick={() => setIsQuickActionOpen(!isQuickActionOpen)}
                        className={`w-16 h-16 rounded-full flex items-center justify-center shadow-xl border-[6px] border-gray-50 dark:border-slate-950 transition-all duration-300 z-40 ${
                          isQuickActionOpen ? 'bg-slate-800 dark:bg-slate-700 rotate-45 scale-95' : 'bg-blue-600 hover:scale-105 hover:bg-blue-700'
                        }`}
                      >
                        <Icon name="Plus" size={32} className="text-white" strokeWidth={2.5} />
                      </button>
                    </div>

                    <div className="flex space-x-6">
                      <button onClick={() => handleTabChange('Progress')} className={`flex flex-col items-center p-2 transition-colors ${activeTab === 'Progress' ? 'text-blue-600 dark:text-blue-400' : 'text-slate-400 hover:text-slate-600 dark:hover:text-slate-300'}`}>
                        <Icon name="TrendingUp" size={24} strokeWidth={activeTab === 'Progress' ? 2.5 : 2} />
                        <span className="text-[10px] mt-1.5 font-bold">Progress</span>
                      </button>
                      <button onClick={() => handleTabChange('Profil')} className={`flex flex-col items-center p-2 transition-colors ${activeTab === 'Profil' ? 'text-blue-600 dark:text-blue-400' : 'text-slate-400 hover:text-slate-600 dark:hover:text-slate-300'}`}>
                        <Icon name="User" size={24} strokeWidth={activeTab === 'Profil' ? 2.5 : 2} />
                        <span className="text-[10px] mt-1.5 font-bold">Profil</span>
                      </button>
                    </div>

                  </div>
                </nav>

                {/* MODAL: Calendar Picker */}
                {isCalendarOpen && (
                  <div className="absolute inset-0 bg-gray-50/90 dark:bg-slate-950/90 backdrop-blur-sm z-50 flex items-center justify-center animate-in fade-in duration-200">
                    <div className="bg-white dark:bg-slate-900 w-[90%] max-w-sm rounded-[2rem] p-6 shadow-2xl border border-gray-100 dark:border-slate-800" onClick={e => e.stopPropagation()}>
                      <div className="flex justify-between items-center mb-6">
                         <button
                           onClick={() => setCalendarDate(new Date(calendarDate.getFullYear(), calendarDate.getMonth() - 1, 1))}
                           className="p-2 hover:bg-slate-100 dark:hover:bg-slate-800 rounded-full transition-colors"
                         >
                           <Icon name="ChevronLeft" size={20} />
                         </button>
                         <h3 className="font-bold text-lg text-slate-800 dark:text-slate-200">
                           {calendarDate.toLocaleDateString('de-DE', { month: 'long', year: 'numeric' })}
                         </h3>
                         <button
                           onClick={() => setCalendarDate(new Date(calendarDate.getFullYear(), calendarDate.getMonth() + 1, 1))}
                           className="p-2 hover:bg-slate-100 dark:hover:bg-slate-800 rounded-full transition-colors"
                         >
                           <Icon name="ChevronRight" size={20} />
                         </button>
                      </div>

                      <div className="grid grid-cols-7 gap-1 mb-2">
                        {['Mo', 'Di', 'Mi', 'Do', 'Fr', 'Sa', 'So'].map(day => (
                          <div key={day} className="text-center text-xs font-semibold text-slate-400 dark:text-slate-500 py-2">
                            {day}
                          </div>
                        ))}
                      </div>

                      <div className="grid grid-cols-7 gap-1">
                        {generateCalendarDays(calendarDate).map((date, i) => {
                          const isToday = date && date.getDate() === new Date().getDate() && date.getMonth() === new Date().getMonth();
                          const isSelectedDay = date && date.getDate() === parseInt(currentDay.date) && currentDay.monthYear.includes(date.getFullYear().toString());

                          return (
                            <button
                              key={i}
                              disabled={!date}
                              onClick={() => handleCalendarDateSelect(date)}
                              className={`
                                aspect-square flex items-center justify-center rounded-xl text-sm font-medium transition-all
                                ${!date ? 'opacity-0 cursor-default' : 'hover:bg-slate-100 dark:hover:bg-slate-800'}
                                ${isToday && !isSelectedDay ? 'text-blue-600 dark:text-blue-400 font-bold' : 'text-slate-700 dark:text-slate-300'}
                                ${isSelectedDay && date ? 'bg-blue-600 text-white hover:bg-blue-700 dark:hover:bg-blue-600 shadow-md shadow-blue-600/20' : ''}
                              `}
                            >
                              {date ? date.getDate() : ''}
                            </button>
                          );
                        })}
                      </div>

                      <button
                        onClick={() => setIsCalendarOpen(false)}
                        className="w-full mt-6 py-3 bg-slate-100 dark:bg-slate-800 text-slate-700 dark:text-slate-300 rounded-xl font-bold hover:bg-slate-200 dark:hover:bg-slate-700 transition-colors"
                      >
                        Schließen
                      </button>
                    </div>
                  </div>
                )}

                {/* MODAL: Training Loggen */}
                {isTrainingLogModalOpen && (
                  <div className="absolute inset-0 bg-gray-50 dark:bg-slate-950 z-50 flex flex-col animate-in slide-in-from-bottom-full duration-300 text-slate-800 dark:text-slate-200">
                    <header className="px-6 pt-12 pb-4 flex justify-between items-center relative">
                      <button
                         onClick={() => {
                             if (isExerciseInfoOpen) {
                                 setIsExerciseInfoOpen(false);
                             } else if (selectedExerciseIndex !== null) {
                                 setSelectedExerciseIndex(null);
                             } else {
                                 setIsTrainingLogModalOpen(false);
                             }
                         }}
                         className="p-2 text-slate-500 dark:text-slate-300 hover:text-slate-800 dark:hover:text-white hover:bg-slate-200 dark:hover:bg-slate-800 rounded-full transition-colors">
                        <Icon name="ChevronLeft" size={24} />
                      </button>
                      <div className="text-center">
                        <h2 className="text-lg font-bold text-slate-900 dark:text-white">
                            {isExerciseInfoOpen ? `Info: ${todaysExercises[selectedExerciseIndex]?.name}` :
                             selectedExerciseIndex !== null ? todaysExercises[selectedExerciseIndex].name :
                             `${currentDay.type} + Acc`}
                        </h2>
                        {selectedExerciseIndex === null && !isExerciseInfoOpen ? (
                             <p className="text-xs text-slate-500 dark:text-slate-400">{todaysExercises.length} Übungen</p>
                        ) : (isWorkoutStarted || isTrainingLogged) && !isExerciseInfoOpen ? (
                             <p className={`text-xs font-bold font-mono ${isTrainingLogged ? 'text-emerald-600 dark:text-emerald-400' : 'text-blue-600 dark:text-blue-400'}`}>{formatTime(elapsedTime)}</p>
                        ) : null}
                      </div>
                      <div className="flex gap-2">
                        {(isWorkoutStarted || isTrainingLogged) && selectedExerciseIndex === null && !isExerciseInfoOpen && (
                            <div className={`flex items-center justify-center px-3 py-1 rounded-full border ${isTrainingLogged ? 'bg-emerald-50 dark:bg-emerald-900/30 border-emerald-200 dark:border-emerald-800' : 'bg-blue-50 dark:bg-blue-900/30 border-blue-200 dark:border-blue-800'}`}>
                                <span className={`text-xs font-bold font-mono ${isTrainingLogged ? 'text-emerald-600 dark:text-emerald-400' : 'text-blue-600 dark:text-blue-400'}`}>{formatTime(elapsedTime)}</span>
                            </div>
                        )}
                        <button className="p-2 text-slate-500 dark:text-slate-300 hover:text-slate-800 dark:hover:text-white hover:bg-slate-200 dark:hover:bg-slate-800 rounded-full transition-colors">
                          <Icon name="RefreshCw" size={20} />
                        </button>
                        <button
                          onClick={() => {
                              if (selectedExerciseIndex === null) {
                                  // Navigiere zum Plan
                                  setIsTrainingLogModalOpen(false);
                                  handleTabChange('Plan');
                                  setActivePlanView('TrainingPlan');
                              } else {
                                  // Toggle Info-Screen für Übung
                                  setIsExerciseInfoOpen(!isExerciseInfoOpen);
                              }
                          }}
                          className={`p-2 rounded-full transition-colors ${isExerciseInfoOpen ? 'bg-blue-100 dark:bg-blue-900 text-blue-600 dark:text-blue-400' : 'text-slate-500 dark:text-slate-300 hover:text-slate-800 dark:hover:text-white hover:bg-slate-200 dark:hover:bg-slate-800'}`}>
                          <Icon name="Info" size={20} />
                        </button>
                      </div>
                    </header>

                    {isExerciseInfoOpen ? (
                      // --- ÜBUNGS-INFO / PLACEHOLDER ---
                      <div className="flex-1 flex flex-col items-center justify-center p-6 text-center animate-in fade-in duration-300">
                          <div className="w-20 h-20 bg-slate-100 dark:bg-slate-800 rounded-full flex items-center justify-center mb-6 shadow-inner">
                              <Icon name="PlayCircle" size={40} className="text-slate-400" />
                          </div>
                          <h3 className="text-xl font-bold text-slate-900 dark:text-white mb-2">{todaysExercises[selectedExerciseIndex]?.name}</h3>
                          <p className="text-slate-500 dark:text-slate-400 mb-8 max-w-xs">
                              Hier könnte ein Erklärvideo, eine Bild-Anleitung oder detaillierte Ausführungshinweise zu dieser Übung stehen.
                          </p>
                          <button
                              onClick={() => setIsExerciseInfoOpen(false)}
                              className="px-6 py-3 bg-white dark:bg-slate-800 border border-slate-200 dark:border-slate-700 rounded-xl font-semibold text-slate-700 dark:text-slate-300 hover:bg-slate-50 dark:hover:bg-slate-700 transition-colors shadow-sm"
                          >
                              Zurück zum Logging
                          </button>
                      </div>
                    ) : selectedExerciseIndex === null ? (
                      // --- ÜBUNGS-ÜBERSICHT ---
                      <div className="flex-1 overflow-y-auto px-4 pb-24">
                         {/* Tabs for "Übungen" and "Ablauf" (Visual only for now) */}
                         <div className="flex border-b border-gray-200 dark:border-slate-800 mb-4 px-2">
                             <div className="flex-1 text-center pb-3 border-b-2 border-blue-600 dark:border-blue-400 text-slate-900 dark:text-white font-medium text-sm">Übungen</div>
                             <div className="flex-1 text-center pb-3 text-slate-500 dark:text-slate-400 font-medium text-sm">Ablauf</div>
                         </div>

                         <div className="space-y-3">
                           {todaysExercises.map((ex, idx) => {
                             // Calculate target rep string for overview
                             const parsedExSets3 = safeArrayParse(ex.plannedSets);
                            const targetReps = parsedExSets3 ? parsedExSets3.map(s => s.reps).join(' / ') : ex.reps;
                            const numSets = parsedExSets3 ? parsedExSets3.length : ex.sets;

                             return (
                               <div
                                 key={idx}
                                 onClick={() => setSelectedExerciseIndex(idx)}
                                 className="bg-white dark:bg-slate-900 p-5 rounded-3xl cursor-pointer hover:bg-gray-50 dark:hover:bg-slate-800 transition-colors border border-gray-100 dark:border-slate-800 shadow-sm flex items-center gap-4"
                               >
                                 <div className="w-12 h-12 bg-blue-50 dark:bg-blue-500/10 text-blue-600 dark:text-blue-400 rounded-2xl flex items-center justify-center">
                                    <Icon name="Dumbbell" size={24} />
                                 </div>
                                 <div>
                                   <h3 className="text-base font-bold text-slate-900 dark:text-white mb-1">{ex.name}</h3>
                                   <p className="text-sm font-medium text-slate-500 dark:text-slate-400">{numSets} Sätze, {targetReps} Wdh.</p>
                                 </div>
                                 <div className="ml-auto text-slate-300 dark:text-slate-600">
                                    <Icon name="ChevronRight" size={20} />
                                 </div>
                               </div>
                             );
                           })}
                         </div>

                         <div className="fixed bottom-8 left-0 w-full flex justify-center px-4">
                            {isTrainingLogged ? (
                                <div className="w-full max-w-[300px] flex gap-3">
                                    <button
                                        onClick={deleteWorkout}
                                        className="flex-1 bg-red-50 text-red-600 dark:bg-red-900/20 dark:text-red-400 border border-red-200 dark:border-red-800 hover:bg-red-100 dark:hover:bg-red-900/40 font-semibold py-3.5 rounded-xl transition-colors"
                                    >
                                        Löschen
                                    </button>
                                    <button
                                        onClick={handleFinishWorkout}
                                        className="flex-1 bg-blue-600 hover:bg-blue-700 text-white font-semibold py-3.5 rounded-xl shadow-lg shadow-blue-600/20 transition-colors"
                                    >
                                        Speichern
                                    </button>
                                </div>
                            ) : !isWorkoutStarted ? (
                                <button
                                    onClick={startWorkout}
                                    className="w-full max-w-[300px] bg-emerald-500 hover:bg-emerald-600 text-white font-semibold py-3.5 rounded-xl shadow-lg shadow-emerald-500/20 transition-colors flex items-center justify-center gap-2"
                                >
                                    <Icon name="Play" size={20} className="fill-current" />
                                    Einheit starten
                                </button>
                            ) : (
                                <button
                                    onClick={handleFinishWorkout}
                                    className="w-full max-w-[300px] bg-blue-600 hover:bg-blue-700 text-white font-semibold py-3.5 rounded-xl shadow-lg shadow-blue-600/20 transition-colors"
                                >
                                    Einheit abschließen
                                </button>
                            )}
                         </div>
                      </div>
                    ) : (
                      // --- ÜBUNGS-DETAILS (Interactive Form) ---
                      <div className="flex-1 overflow-y-auto px-4 pb-24">

                          {/* Grid Header */}
                          {/* Coach Note (if available) */}
                          {todaysExercises[selectedExerciseIndex]?.note && (
                              <div className="bg-blue-50 dark:bg-blue-900/20 border border-blue-200 dark:border-blue-800/50 rounded-2xl p-4 mb-4 flex gap-3 shadow-sm">
                                  <Icon name="Info" size={20} className="text-blue-500 shrink-0 mt-0.5" />
                                  <div>
                                      <h4 className="text-[10px] font-bold text-blue-500 uppercase tracking-widest mb-1">Coach Note</h4>
                                      <p className="text-sm font-medium text-slate-700 dark:text-slate-300">
                                          {todaysExercises[selectedExerciseIndex].note}
                                      </p>
                                  </div>
                              </div>
                          )}

                          {/* Grid Header */}
                          <div className="flex px-3 text-xs font-bold text-slate-400 dark:text-slate-500 mb-3 items-end">
                              <div className="flex-1 grid grid-cols-[1fr_1fr_1fr_36px] gap-2">
                                  <div>
                                     <span className="block text-[10px] uppercase tracking-wider">Gewicht</span>
                                     <span className="block text-slate-600 dark:text-slate-300 font-extrabold">kg</span>
                                  </div>
                                  <div className="text-center">
                                     <span className="block text-[10px] uppercase tracking-wider">Wdh.</span>
                                     <span className="block text-slate-600 dark:text-slate-300 font-extrabold">#</span>
                                  </div>
                                  <div className="text-center">
                                     <span className="block text-[10px] uppercase tracking-wider">RPE</span>
                                     <span className="block text-slate-600 dark:text-slate-300 font-extrabold">RPE</span>
                                  </div>
                                  <div></div>
                              </div>
                              <div className="w-14"></div>
                          </div>

                          <div className="space-y-3">
                              {/* Render sets. First ensure state exists for this exercise */}
                              {(() => {
                                  const dayKey = currentDay.fullDate || dateStr;
                                  let setsToRender = trainingLogState[dayKey] && trainingLogState[dayKey][selectedExerciseIndex];
                                  if (!setsToRender) {
                                      // Default initialization if state isn't there yet
                                      const ex = todaysExercises[selectedExerciseIndex];
                                      const parsedExSets4 = safeArrayParse(ex.plannedSets);
                                     setsToRender = parsedExSets4 ?
                                         parsedExSets4.map(s => ({ ...s, isCompleted: false, actualWeight: '', actualReps: '', actualRpe: '', videoUrl: '' })) :
                                          Array.from({ length: ex.sets }, () => ({ weight: '', reps: ex.reps, rpe: ex.rpe, videoRequested: false, isCompleted: false, actualWeight: '', actualReps: '', actualRpe: '', videoUrl: '' }));
                                  }

                                  const setsList = setsToRender.map((setObj, setIdx) => {
                                      // Logik zum automatischen Aktivieren des "Vollständig"-Checkmarks
                                      const isActualWeightFilled = !!setObj.actualWeight || !!setObj.plannedWeight;
                                      const isActualRepsFilled = !!setObj.actualReps;
                                      const isActualRpeFilled = !!setObj.actualRpe;
                                      const isSetCompleteable = isActualWeightFilled && isActualRepsFilled && isActualRpeFilled;

                                      return (
                                          <div key={setIdx} className={`bg-white dark:bg-slate-900 rounded-3xl flex overflow-hidden border shadow-sm ${setObj.isCompleted ? 'border-blue-500 opacity-90' : 'border-gray-100 dark:border-slate-800'} transition-all relative`}>

                                              {/* Left Section (Inputs) */}
                                              <div className="flex-1 grid grid-cols-[1fr_1fr_1fr_36px] gap-2 p-3 items-center">

                                                  {/* Weight Input */}
                                                  <div className="flex flex-col">
                                                      <span className="text-[10px] font-semibold text-slate-400 dark:text-slate-500 ml-1 mb-1">{setObj.plannedWeight ? setObj.plannedWeight : (setObj.previousWeight ? `(${setObj.previousWeight})` : '-')}</span>
                                                      <input
                                                          type="number"
                                                          value={setObj.plannedWeight ? setObj.plannedWeight : setObj.actualWeight}
                                                          onChange={(e) => {
                                                              if (!setObj.plannedWeight) {
                                                                  updateSetInput(selectedExerciseIndex, setIdx, 'actualWeight', e.target.value);
                                                              }
                                                          }}
                                                          placeholder={setObj.previousWeight ? setObj.previousWeight : setObj.plannedWeight || setObj.weight}
                                                          disabled={!!setObj.plannedWeight || setObj.isCompleted}
                                                          className={`w-full bg-transparent border-b-2 border-gray-200 dark:border-slate-700 focus:border-blue-500 text-slate-900 dark:text-white font-bold text-center pb-1 outline-none transition-colors ${(!!setObj.plannedWeight || setObj.isCompleted) ? 'text-slate-500 dark:text-slate-400 border-transparent cursor-not-allowed' : ''}`}
                                                      />
                                                  </div>

                                                  {/* Reps Input */}
                                                  <div className="flex flex-col text-center">
                                                      <span className="text-[10px] font-semibold text-slate-400 dark:text-slate-500 ml-1 mb-1">{setObj.reps ? setObj.reps : '-'}</span>
                                                      <input
                                                          type="number"
                                                          value={setObj.actualReps}
                                                          onChange={(e) => updateSetInput(selectedExerciseIndex, setIdx, 'actualReps', e.target.value)}
                                                          disabled={setObj.isCompleted}
                                                          className={`w-full bg-transparent border-b-2 border-gray-200 dark:border-slate-700 focus:border-blue-500 text-slate-900 dark:text-white font-bold text-center pb-1 outline-none transition-colors ${setObj.isCompleted ? 'text-slate-500 dark:text-slate-400 border-transparent cursor-not-allowed' : ''}`}
                                                      />
                                                  </div>

                                                  {/* RPE Input */}
                                                  <div className="flex flex-col text-center">
                                                      <span className="text-[10px] font-semibold text-transparent select-none ml-1 mb-1">-</span>
                                                      <input
                                                          type="number"
                                                          step="0.5"
                                                          value={setObj.actualRpe}
                                                          onChange={(e) => updateSetInput(selectedExerciseIndex, setIdx, 'actualRpe', e.target.value)}
                                                          disabled={setObj.isCompleted}
                                                          className={`w-full bg-transparent border-b-2 border-gray-200 dark:border-slate-700 focus:border-blue-500 text-slate-900 dark:text-white font-bold text-center pb-1 outline-none transition-colors ${setObj.isCompleted ? 'text-slate-500 dark:text-slate-400 border-transparent cursor-not-allowed' : ''}`}
                                                      />
                                                  </div>

                                                  {/* Video Button */}
                                                  <div className="flex justify-center mt-4 relative">
                                                      <button
                                                          onClick={() => openVideoModal(selectedExerciseIndex, setIdx)}
                                                          className={`transition-colors p-1.5 rounded-lg ${
                                                              setObj.videoUrl
                                                                  ? 'bg-emerald-100 dark:bg-emerald-500/20 text-emerald-600 dark:text-emerald-400'
                                                                  : setObj.videoRequested
                                                                      ? 'bg-blue-100 dark:bg-blue-500/20 text-blue-600 dark:text-blue-400 animate-pulse'
                                                                      : 'text-slate-400 hover:text-slate-600 dark:text-slate-500 dark:hover:text-slate-300'
                                                          }`}
                                                      >
                                                          <Icon name={setObj.videoUrl ? "CheckCircle2" : "Video"} size={20} />
                                                      </button>
                                                  </div>

                                              </div>

                                              {/* Right Section (Check Button) */}
                                              <button
                                                  onClick={() => {
                                                      if (isSetCompleteable || setObj.isCompleted) {
                                                          if (!isWorkoutStarted && !isTrainingLogged) {
                                                              startWorkout(); // Einheit automatisch starten, wenn nicht aktiv und noch nicht abgeschlossen
                                                          }
                                                          toggleSetCompleted(selectedExerciseIndex, setIdx);
                                                      }
                                                  }}
                                                  disabled={!isSetCompleteable && !setObj.isCompleted}
                                                  className={`w-14 flex items-center justify-center transition-colors ${setObj.isCompleted ? 'bg-blue-600 text-white cursor-pointer' : isSetCompleteable ? 'bg-blue-100 text-blue-500 dark:bg-blue-900/30 hover:bg-blue-200 cursor-pointer animate-pulse' : 'bg-gray-100 dark:bg-slate-800 text-slate-300 dark:text-slate-600 cursor-not-allowed'}`}
                                              >
                                                  <Icon name="Check" size={24} />
                                              </button>
                                          </div>
                                      );
                                  });

                                  // Add note input at the end of the sets
                                  return (
                                      <>
                                          {setsList}
                                          <div className={`mt-4 bg-white dark:bg-slate-900 rounded-3xl p-3 border border-gray-100 dark:border-slate-800 shadow-sm relative`}>
                                              <textarea
                                                  placeholder="Notiz zu dieser Übung hinzufügen..."
                                                  className="w-full bg-transparent text-sm text-slate-700 dark:text-slate-300 outline-none resize-none"
                                                  rows="2"
                                                  value={exerciseNotes[dayKey]?.[selectedExerciseIndex] || ''}
                                                  onChange={(e) => setExerciseNotes(prev => {
                                                      const newState = {...prev};
                                                      if (!newState[dayKey]) newState[dayKey] = {};
                                                      newState[dayKey][selectedExerciseIndex] = e.target.value;
                                                      return newState;
                                                  })}
                                              ></textarea>
                                          </div>
                                      </>
                                  );
                              })()}
                          </div>

                          <div className="fixed bottom-8 left-0 w-full flex justify-center px-4 z-10">
                              <button
                                  onClick={() => setSelectedExerciseIndex(null)}
                                  className="w-full max-w-[300px] bg-blue-600 hover:bg-blue-700 text-white font-semibold py-3.5 rounded-xl shadow-lg shadow-blue-600/20 transition-colors"
                              >
                                  Übung abschließen
                              </button>
                          </div>

                          {/* Floating Rest Timer Component */}
                          {(restTimer.isActive || restTimer.totalSeconds > 0 || restTimer.isFlashing) && (
                              <div
                                  className={`fixed bottom-28 right-6 z-50 flex items-center justify-center cursor-pointer transition-all duration-300 ${restTimer.isFlashing ? 'animate-pulse scale-110' : ''}`}
                                  onClick={(e) => {
                                      if (e.detail === 2) {
                                          handleRestTimerClick(e);
                                      } else {
                                          if (!restTimer.isActive && !restTimer.isFlashing) {
                                              handleManualRestTimeEdit();
                                          } else {
                                              handleRestTimerClick(e);
                                          }
                                      }
                                  }}
                              >
                                  <div className="relative w-16 h-16 flex items-center justify-center bg-white dark:bg-slate-900 rounded-full shadow-2xl border border-slate-200 dark:border-slate-800">
                                      <svg className="absolute w-full h-full transform -rotate-90">
                                          <circle cx="32" cy="32" r="28" stroke="currentColor" strokeWidth="4" fill="transparent" className="text-slate-100 dark:text-slate-800" />
                                          {restTimer.totalSeconds > 0 && !restTimer.isFlashing && (
                                              <circle
                                                  cx="32" cy="32" r="28"
                                                  stroke="currentColor" strokeWidth="4" fill="transparent"
                                                  strokeDasharray="176"
                                                  strokeDashoffset={176 - (176 * (restTimer.secondsLeft / restTimer.totalSeconds))}
                                                  strokeLinecap="round"
                                                  className={`${restTimer.isPaused ? 'text-amber-500' : 'text-blue-500'} transition-all duration-1000 ease-linear`}
                                              />
                                          )}
                                          {restTimer.isFlashing && (
                                              <circle
                                                  cx="32" cy="32" r="28"
                                                  stroke="currentColor" strokeWidth="4" fill="transparent"
                                                  className="text-red-500 animate-pulse"
                                              />
                                          )}
                                      </svg>
                                      <div className="absolute flex flex-col items-center justify-center w-full h-full">
                                          <span className={`text-sm font-black mt-1 ${restTimer.isFlashing ? 'text-red-500' : restTimer.isPaused ? 'text-amber-500 dark:text-amber-400' : 'text-slate-800 dark:text-slate-100'}`}>
                                              {restTimer.isFlashing ? '0:00' : `${Math.floor(restTimer.secondsLeft / 60)}:${(restTimer.secondsLeft % 60).toString().padStart(2, '0')}`}
                                          </span>
                                          {restTimer.isPaused && <span className="text-[8px] font-bold text-amber-500 uppercase mt-[-2px]">PAUSED</span>}
                                      </div>
                                  </div>
                              </div>
                          )}

                      </div>
                    )}

                    {/* VIDEO UPLOAD MODAL */}
                    {videoModalConfig.isOpen && (
                        <div className="absolute inset-0 bg-black/80 z-[60] flex items-center justify-center p-4 backdrop-blur-sm animate-in fade-in">
                            <div className="bg-[#1e2329] w-full max-w-sm rounded-[2rem] p-6 shadow-2xl border border-slate-700/50">
                                <div className="flex justify-between items-center mb-6">
                                    <h3 className="text-lg font-bold text-white flex items-center gap-2">
                                        <Icon name="Video" size={20} className="text-blue-400" />
                                        Video Link
                                    </h3>
                                    <button
                                        onClick={() => setVideoModalConfig({ isOpen: false, exerciseIndex: null, setIndex: null })}
                                        className="text-slate-400 hover:text-white transition-colors"
                                    >
                                        <Icon name="X" size={20} />
                                    </button>
                                </div>
                                <p className="text-sm text-slate-400 mb-4">
                                    Füge hier den Link zu deinem Trainingsvideo ein (z.B. YouTube, Google Drive).
                                </p>
                                <input
                                    type="text"
                                    value={tempVideoUrl}
                                    onChange={(e) => setTempVideoUrl(e.target.value)}
                                    placeholder="https://..."
                                    className="w-full bg-[#282e35] text-white border border-slate-700 focus:border-blue-500 rounded-xl px-4 py-3 outline-none transition-colors mb-6"
                                />
                                <div className="flex gap-3">
                                    <button
                                        onClick={() => setVideoModalConfig({ isOpen: false, exerciseIndex: null, setIndex: null })}
                                        className="flex-1 py-3 rounded-xl font-medium text-slate-300 hover:bg-[#282e35] transition-colors"
                                    >
                                        Abbrechen
                                    </button>
                                    <button
                                        onClick={saveVideoUrl}
                                        className="flex-1 py-3 rounded-xl font-medium bg-blue-600 hover:bg-blue-700 text-white transition-colors shadow-lg shadow-blue-600/20"
                                    >
                                        Speichern
                                    </button>
                                </div>
                            </div>
                        </div>
                    )}
                  </div>
                )}

              </div>
            </div>
          );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
