const calculateTargetWeight = (weightInput, exName, repsInput, rpeInput, athlete1RMs) => {
    if (!weightInput) return null;
    const isDynamic = String(weightInput).trim().toLowerCase() === "1rm" || String(weightInput).trim().toLowerCase() === "sess1rm";
    if (!isDynamic) return null;

    if (!athlete1RMs || !Array.isArray(athlete1RMs)) return null;
    const stored1RMObj = athlete1RMs.find(rm => rm.exercise && rm.exercise.toLowerCase() === (exName || '').toLowerCase());
    const base1RM = stored1RMObj && stored1RMObj.oneRm ? Number(stored1RMObj.oneRm) : null;
    if (!base1RM) return null;

    let parsedReps = 1;
    if (repsInput) {
        const parts = String(repsInput).split('-');
        const val = parseInt(parts[parts.length - 1], 10);
        if (!isNaN(val)) parsedReps = val;
    }
    if (parsedReps > 15) parsedReps = 15;
    if (parsedReps < 1) parsedReps = 1;

    let parsedRPE = 10;
    if (rpeInput) {
        const rpeVal = parseFloat(String(rpeInput).replace(',', '.'));
        if (!isNaN(rpeVal)) parsedRPE = rpeVal;
    }
    
    const validRPEs = [10, 9.5, 9, 8.5, 8, 7.5, 7, 6.5, 6];
    let closestRPE = 10;
    let minDiff = Infinity;
    for (let r of validRPEs) {
        if (Math.abs(r - parsedRPE) < minDiff) {
            minDiff = Math.abs(r - parsedRPE);
            closestRPE = r;
        }
    }

    const percentage = RPE_CHART[parsedReps] ? RPE_CHART[parsedReps][closestRPE] : null;
    if (!percentage) return null;

    const calculated = (base1RM * percentage) / 100;
    return Math.round(calculated * 2) / 2; // nearest 0.5kg
};

        // Custom Component für Icons, da lucide-react in diesem Setup nicht direkt funktioniert
        // Wir nutzen das globale "lucide" Objekt, um SVG Strings zu generieren und rendern diese via dangerouslySetInnerHTML
        const Icon = ({ name, size = 24, className = '', strokeWidth = 2 }) => {
            // Lucide Icons global keys are PascalCase (e.g. CheckCircle2)
            // But sometimes might be requested by the component differently. We just map straight to the name.
            let iconName = name;

            // Allow matching names properly with or without case difference
            // Also replace dashes which are often requested in standard lucide React usage
            const normalizedRequestedName = name.replace(/-/g, '').toLowerCase();

            if (!lucide.icons[iconName]) {
                 const match = Object.keys(lucide.icons).find(k => k.toLowerCase() === normalizedRequestedName);
                 if (match) {
                     iconName = match;
                 }
            }

            // Fallback, falls Icon nicht gefunden
            if (!lucide.icons[iconName]) {
                 console.warn(`Icon ${name} not found in lucide.`);
                 return <span className={className} style={{ width: size, height: size, display: 'inline-block' }}></span>;
            }

            // Build the SVG using React elements since toSvg() is deprecated/removed in some unpkg bundles
            const iconData = lucide.icons[iconName];

            const children = iconData.map(([tag, attrs], index) => {
                return React.createElement(tag, { ...attrs, key: index });
            });

            return React.createElement(
                'svg',
                {
                    width: size,
                    height: size,
                    viewBox: "0 0 24 24",
                    fill: "none",
                    stroke: "currentColor",
                    strokeWidth: strokeWidth,
                    strokeLinecap: "round",
                    strokeLinejoin: "round",
                    className: `lucide lucide-${iconName} ${className}`
                },
                children
            );
        };

        if(athlete) {
            // Load Cycles
            if (typeof google !== 'undefined' && google.script && google.script.run) {
                google.script.run
                    .withSuccessHandler((res) => {
                        if(res && res.success) setSavedCycles(res.data || []);
                    })
                    .loadCyclesByAthlete(athlete.id || athlete.name);
                    
                // Load Meta Data
                setIsLoadingMeta(true);
                google.script.run
                    .withSuccessHandler((res) => {
                        if(res && res.success && res.data) {
                            if(res.data.weightHistory && res.data.weightHistory.length > 0) setWeightHistory(res.data.weightHistory);
                            if(res.data.oneRMs) setStrengthData(res.data.oneRMs);
                        }
                        setIsLoadingMeta(false);
                    })
                    .getAthleteMetaData(athlete.id);

                 setSavedCycles([
                    { id: 'cycle-mock', name: 'Hypertrophie Block 1', startDate: '2023-10-01' }

                setPlans(plans.map(p => {
                    if (p.id !== draggingPlan.id) return p;
                    let newStart = p.startWeek;
                    let newDuration = p.durationWeeks;

                    if (draggingPlan.edge === 'left') {
                        const newStartFloat = draggingPlan.originalStart + weeksDelta;
                        const maxStart = draggingPlan.originalStart + draggingPlan.originalDuration - 0.5;
                        newStart = Math.max(1, Math.min(newStartFloat, maxStart));
                        newDuration = draggingPlan.originalDuration - (newStart - draggingPlan.originalStart);
                    } else if (draggingPlan.edge === 'right') {
                        const newDurationFloat = draggingPlan.originalDuration + weeksDelta;
                        newDuration = Math.max(0.5, newDurationFloat);
                    } else if (draggingPlan.edge === 'body') {
                        const newStartFloat = draggingPlan.originalStart + weeksDelta;
                        newStart = Math.max(1, newStartFloat);
                    }

                    // Snap to half weeks
                    newStart = Math.round(newStart * 2) / 2;
                    newDuration = Math.round(newDuration * 2) / 2;

                    const newEnd = newStart + newDuration - 1;

                    // Auto-extend block logic is triggered when dragging ends or via an effect.
                    // Doing it repeatedly during pointer move causes massive re-renders and glitching.

                    return { ...p, startWeek: newStart, durationWeeks: newDuration };
                }));
            };

            if (draggingPlan.type === 'training') updatePlans(trainingPlans, setTrainingPlans);
            else updatePlans(nutritionPlans, setNutritionPlans);
            return;
        }

        if (!draggingBlock) return;

        setBlocks(prevBlocks => prevBlocks.map(block => {
            if (block.id !== draggingBlock.id) return block;

            // Simple pixel to value mapping
            if (draggingBlock.type === 'duration') {
                const deltaX = e.clientX - draggingBlock.startX;
                // Assuming roughly 50px = 1 week
                const weeksDelta = Math.round(deltaX / 50);
                let newDuration = Math.max(1, draggingBlock.originalDuration + weeksDelta); // min 1 week

                // Recalculate weekly data if duration changes
                let newWeeklyData = [...block.weeklyData];
                if (newDuration > newWeeklyData.length) {
                    for(let i = newWeeklyData.length; i < newDuration; i++) {
                        newWeeklyData.push({ week: block.startWeek + i, volume: block.volume, intensity: block.intensity });
                    }
                } else if (newDuration < newWeeklyData.length) {
                    newWeeklyData = newWeeklyData.slice(0, newDuration);
                }

                return { ...block, durationWeeks: newDuration, weeklyData: newWeeklyData };
            }
            else if (draggingBlock.type === 'height') {
                const deltaY = draggingBlock.startY - e.clientY; // Up is positive
                // 2px = 1%
                const percentDelta = Math.round(deltaY / 2);
                let newValue = Math.max(0, Math.min(100, draggingBlock.originalValue + percentDelta));

                if (timelineView === 'Volumen') {
                    const updatedWeekly = block.weeklyData.map(w => ({...w, volume: newValue}));
                    return { ...block, volume: newValue, weeklyData: updatedWeekly };
                } else {
                    const updatedWeekly = block.weeklyData.map(w => ({...w, intensity: newValue}));
                    return { ...block, intensity: newValue, weeklyData: updatedWeekly };
                }
            }
            else if (draggingBlock.type === 'weeklyPoint') {
                const containerHeight = currentTargetRect ? currentTargetRect.height : 300; // fallback height
                const deltaY = draggingBlock.startY - e.clientY;
                // Rough estimate, let's use the same 2px = 1% for now, or calculate based on container
                const percentDelta = Math.round(deltaY / 2);
                let newValue = Math.max(0, Math.min(100, draggingBlock.originalValue + percentDelta));

                let newWeeklyData = [...block.weeklyData];
                if (timelineView === 'Volumen') {
                    newWeeklyData[draggingBlock.weekIndex] = { ...newWeeklyData[draggingBlock.weekIndex], volume: newValue };
                    // Recalculate block average
                    const avgVol = Math.round(newWeeklyData.reduce((sum, w) => sum + w.volume, 0) / newWeeklyData.length);
                    return { ...block, volume: avgVol, weeklyData: newWeeklyData };
                } else {
                    newWeeklyData[draggingBlock.weekIndex] = { ...newWeeklyData[draggingBlock.weekIndex], intensity: newValue };
                    // Recalculate block average
                    const avgInt = Math.round(newWeeklyData.reduce((sum, w) => sum + w.intensity, 0) / newWeeklyData.length);
                    return { ...block, intensity: avgInt, weeklyData: newWeeklyData };
                }
            }
            return block;
        }));
    };

    const handlePointerUp = (e) => {
        if (draggingPlan) {
            e.target.releasePointerCapture(e.pointerId);
            setDraggingPlan(null);

        }
        if (draggingBlock) {
            e.target.releasePointerCapture(e.pointerId);
            setDraggingBlock(null);

            // Re-calculate subsequent block start weeks based on new durations
            setBlocks(prevBlocks => {
                let currentStart = 1;
                return prevBlocks.map(b => {
                    const newB = { ...b, startWeek: currentStart };

                    // Update weeks internal numbering
                    newB.weeklyData = newB.weeklyData.map((w, i) => ({...w, week: currentStart + i}));

                    currentStart += b.durationWeeks;
                    return newB;
                });
            });
        }
    };

    // -- Render Helpers --
    const totalTimelineWeeks = Math.max(12, durationWeeks + 2); // Show at least 12 weeks, plus padding

    return (
        <div className="p-6 h-full flex flex-col overflow-hidden bg-slate-50 dark:bg-slate-900/50"
             onPointerMove={handlePointerMove}
             onPointerUp={handlePointerUp}>

            {/* Header Actions */}
            <div className="flex items-center justify-between mb-6">
                 <div className="flex items-center gap-4">
                    <button className="px-4 py-2 text-sm font-bold bg-white dark:bg-slate-800 border border-slate-200 dark:border-slate-700 text-slate-800 dark:text-white rounded-xl shadow-sm flex items-center gap-2">
                        <div className={`w-3 h-3 rounded-full ${colors[cycleColor]}`}></div>
                        {cycleName}
                    </button>
                    {/* Color Picker Dropdown Mock */}
                    <div className="flex gap-1">
                        {Object.keys(colors).map(c => (
                            <button key={c} onClick={() => setCycleColor(c)} className={`w-6 h-6 rounded-full ${colors[c]} ${cycleColor === c ? 'ring-2 ring-offset-2 ring-slate-400 dark:ring-slate-600 dark:ring-offset-slate-900' : 'opacity-50 hover:opacity-100'}`}></button>
                        ))}
                    </div>
                 </div>
                 <div className="flex items-center gap-3">

    <select
        value={selectedCycleId}
        onChange={handleLoadCycle}
        className="px-4 py-2 text-sm bg-white dark:bg-slate-800 border border-slate-200 dark:border-slate-700 rounded-xl outline-none mr-4"
    >
        <option value="">-- Zyklus laden --</option>
        {savedCycles.map(c => (
            <option key={c.id} value={c.id}>{c.name} ({c.startDate || 'Draft'})</option>
        ))}
    </select>

    {selectedCycleId && (
        <button onClick={() => {
            if(confirm('Zyklus wirklich archivieren? Die Trainingspläne bleiben erhalten.')) {
                if (typeof google !== 'undefined') {
                    google.script.run
                        .withSuccessHandler((res) => { alert('Archiviert!'); /* reload logic or clear UI */ setSelectedCycleId(''); setCycleName('Neuer Zyklus'); setBlocks([{id:'b1', name:'Block 1', durationWeeks:4}]); })
                        .archiveCycle(selectedCycleId);
                }
            }
        }} className="px-4 py-2 text-sm font-bold bg-red-100 hover:bg-red-200 text-red-600 rounded-xl shadow-sm transition-colors mr-4">
            Zyklus Archivieren
        </button>
    )}
    <button onClick={handleSaveCycle} className="px-6 py-2 text-sm font-bold bg-indigo-600 hover:bg-indigo-700 text-white rounded-xl shadow-sm transition-colors">
        Zyklus Speichern
    </button>


                 </div>
            </div>

            {/* Cycle Metadata Container */}
            <div className="bg-white dark:bg-slate-800 rounded-2xl border border-slate-200 dark:border-slate-700 p-6 flex flex-col h-full overflow-hidden shadow-sm">
                 <div className="flex flex-wrap gap-8 items-start border-b border-slate-100 dark:border-slate-700 pb-6 mb-6 shrink-0">
                    <div>
                        <div className="text-xs font-bold text-slate-500 dark:text-slate-400 uppercase tracking-wider mb-1 flex items-center gap-1">
                            ZYKLUS NAME
                        </div>
                        <input
                            type="text"
                            value={cycleName}
                            onChange={(e) => setCycleName(e.target.value)}
                            className="text-2xl font-bold text-slate-800 dark:text-white bg-transparent border-b border-transparent hover:border-slate-300 focus:border-blue-500 outline-none w-full"
                        />
                    </div>

                    <div>
                        <div className="text-xs font-bold text-slate-500 dark:text-slate-400 uppercase tracking-wider mb-1 flex items-center gap-1">
                            <Icon name="Target" size={14} /> ZIELE
                        </div>
                        <input
                            type="text"
                            value={cycleGoals}
                            onChange={(e) => setCycleGoals(e.target.value)}
                            placeholder="Ziele eingeben..."
                            className="text-sm font-medium bg-slate-50 dark:bg-slate-900 border border-slate-200 dark:border-slate-700 rounded-lg px-3 py-1.5 outline-none dark:text-white w-64"
                        />
                    </div>

                    <div>
                        <div className="text-xs font-bold text-slate-500 dark:text-slate-400 uppercase tracking-wider mb-1 flex items-center gap-1">
                            <Icon name="Calendar" size={14} /> STARTDATUM
                        </div>
                        <div className="flex items-center gap-2">
                            <input
                                type="date"
                                value={startDate}
                                onChange={(e) => setStartDate(e.target.value)}
                                className="text-sm font-semibold bg-slate-50 dark:bg-slate-900 border border-slate-200 dark:border-slate-700 rounded-lg px-3 py-1.5 outline-none dark:text-white"
                            />

                        </div>
                    </div>

                    <div>
                        <div className="text-xs font-bold text-slate-500 dark:text-slate-400 uppercase tracking-wider mb-1">
                            GESAMTDAUER
                        </div>
                        <div className="flex items-center gap-2 bg-slate-50 dark:bg-slate-900 border border-slate-200 dark:border-slate-700 rounded-lg px-4 py-1.5">
                            <span className="text-xl font-black text-slate-800 dark:text-white">{durationWeeks}</span>
                            <span className="text-sm font-medium text-slate-500">Wochen {startDate && `(bis ${endDateStr})`}</span>
                        </div>
                    </div>
                 </div>

                 {/* Timeline / Calendar Header Controls */}
                 <div className="flex items-center justify-between mb-4 shrink-0">
                    <div className="flex items-center gap-4">
                        <button className="px-3 py-1.5 text-xs font-bold text-red-600 bg-red-50 dark:bg-red-500/10 rounded-lg flex items-center gap-1 border border-red-100 dark:border-red-500/20">
                            + Termin
                        </button>
                        <div className="flex items-center gap-4 ml-4 text-xs font-semibold text-slate-600 dark:text-slate-300">
                            <label className="flex items-center gap-2 cursor-pointer p-1.5 rounded hover:bg-slate-100 dark:hover:bg-slate-700">
                                <input type="checkbox" checked={showProgression} onChange={(e)=>setShowProgression(e.target.checked)} className="rounded border-slate-300" />
                                Progression bearbeiten
                            </label>
                        </div>
                    </div>
                    <div className="flex items-center gap-2 p-1 bg-slate-100 dark:bg-slate-900 rounded-xl">
                        <button
                            onClick={() => setTimelineView('Volumen')}
                            className={`px-4 py-1.5 text-xs font-bold rounded-lg transition-all ${timelineView === 'Volumen' ? 'bg-white dark:bg-slate-800 text-blue-600 shadow-sm' : 'text-slate-500 hover:text-slate-700 dark:text-slate-400'}`}>
                            Volumen
                        </button>
                        <button
                            onClick={() => setTimelineView('Intensität')}
                            className={`px-4 py-1.5 text-xs font-bold rounded-lg transition-all ${timelineView === 'Intensität' ? 'bg-white dark:bg-slate-800 text-red-500 shadow-sm' : 'text-slate-500 hover:text-slate-700 dark:text-slate-400'}`}>
                            Intensität
                        </button>
                    </div>
                 </div>

                 {/* Interactive Gantt Chart / Timeline Area */}
                 <div className="flex-1 border border-slate-200 dark:border-slate-700 rounded-xl overflow-x-auto overflow-y-hidden relative bg-slate-50/50 dark:bg-slate-900/50 custom-scrollbar">

                    {/* Y-Axis Label */}
                    <div className="absolute left-0 top-0 bottom-8 w-10 border-r border-slate-200 dark:border-slate-700 bg-white/80 dark:bg-slate-800/80 backdrop-blur z-20 flex items-center justify-center shadow-sm">
                        <span className="text-[10px] font-bold text-slate-400 tracking-widest -rotate-90 whitespace-nowrap">{timelineView.toUpperCase()} (%)</span>
                    </div>

                    {/* Scrollable Timeline Grid */}
                    <div className="ml-10 h-full flex flex-col relative" style={{ width: `${totalTimelineWeeks * 80}px` }}>

                        {/* Horizontal Background Grid Lines */}
                        <div className="absolute inset-0 flex flex-col justify-between pointer-events-none py-8 z-0">
                            {[100, 75, 50, 25, 0].map(val => (
                                <div key={val} className="w-full border-t border-dashed border-slate-200 dark:border-slate-700/50 relative">
                                     <span className="absolute -left-7 -top-2 text-[8px] text-slate-400">{val}</span>
                                </div>
                            ))}
                        </div>

                        {/* Blocks Layer */}
                        <div className="flex-1 flex pt-8 pb-10 relative z-10">

                            {/* Plans Layer */}
                            {trainingPlans.map(plan => {
                                const leftPercent = ((plan.startWeek - 1) / totalTimelineWeeks) * 100;
                                const widthPercent = (plan.durationWeeks / totalTimelineWeeks) * 100;
                                return (
                                    <div
                                        key={plan.id}
                                        className="absolute bottom-[3rem] h-6 bg-blue-500/80 border border-blue-600 text-white rounded-md flex items-center justify-center text-[10px] font-bold cursor-move group select-none shadow-sm z-30"
                                        style={{ left: `${leftPercent}%`, width: `${widthPercent}%` }}
                                        onPointerDown={(e) => handlePlanPointerDown(e, plan, 'training', 'body')}
                                    >
                                        <div className="absolute left-0 top-0 bottom-0 w-2 cursor-ew-resize hover:bg-black/20" onPointerDown={(e) => handlePlanPointerDown(e, plan, 'training', 'left')}></div>
                                        <input
                                            value={plan.name}
                                            onChange={(e) => setTrainingPlans(trainingPlans.map(p => p.id === plan.id ? {...p, name: e.target.value} : p))}
                                            className="bg-transparent text-center w-full px-2 outline-none pointer-events-auto truncate text-white placeholder-white/70"
                                            onPointerDown={e => e.stopPropagation()}
                                            placeholder="Trainingsplan"
                                        />
                                        <button onClick={() => setTrainingPlans(trainingPlans.filter(p => p.id !== plan.id))} className="absolute -right-1 -top-1 bg-red-500 text-white rounded-full p-0.5 opacity-0 group-hover:opacity-100 pointer-events-auto"><Icon name="X" size={8}/></button>
                                        <div className="absolute right-0 top-0 bottom-0 w-2 cursor-ew-resize hover:bg-black/20" onPointerDown={(e) => handlePlanPointerDown(e, plan, 'training', 'right')}></div>
                                    </div>
                                );
                            })}
                            {nutritionPlans.map(plan => {
                                const leftPercent = ((plan.startWeek - 1) / totalTimelineWeeks) * 100;
                                const widthPercent = (plan.durationWeeks / totalTimelineWeeks) * 100;
                                return (
                                    <div
                                        key={plan.id}
                                        className="absolute bottom-[1.2rem] h-6 bg-emerald-500/80 border border-emerald-600 text-white rounded-md flex items-center justify-center text-[10px] font-bold cursor-move group select-none shadow-sm z-30"
                                        style={{ left: `${leftPercent}%`, width: `${widthPercent}%` }}
                                        onPointerDown={(e) => handlePlanPointerDown(e, plan, 'nutrition', 'body')}
                                    >
                                        <div className="absolute left-0 top-0 bottom-0 w-2 cursor-ew-resize hover:bg-black/20" onPointerDown={(e) => handlePlanPointerDown(e, plan, 'nutrition', 'left')}></div>
                                        <input
                                            value={plan.name}
                                            onChange={(e) => setNutritionPlans(nutritionPlans.map(p => p.id === plan.id ? {...p, name: e.target.value} : p))}
                                            className="bg-transparent text-center w-full px-2 outline-none pointer-events-auto truncate text-white placeholder-white/70"
                                            onPointerDown={e => e.stopPropagation()}
                                            placeholder="Ernährungsplan"
                                        />
                                        <button onClick={() => setNutritionPlans(nutritionPlans.filter(p => p.id !== plan.id))} className="absolute -right-1 -top-1 bg-red-500 text-white rounded-full p-0.5 opacity-0 group-hover:opacity-100 pointer-events-auto"><Icon name="X" size={8}/></button>
                                        <div className="absolute right-0 top-0 bottom-0 w-2 cursor-ew-resize hover:bg-black/20" onPointerDown={(e) => handlePlanPointerDown(e, plan, 'nutrition', 'right')}></div>
                                    </div>
                                );
                            })}

                            {blocks.map((block, idx) => {
                                const leftPercent = ((block.startWeek - 1) / totalTimelineWeeks) * 100;
                                const widthPercent = (block.durationWeeks / totalTimelineWeeks) * 100;

                                const val = timelineView === 'Volumen' ? block.volume : block.intensity;
                                const heightPercent = val; // 0 to 100
                                const colorClass = timelineView === 'Volumen' ? 'bg-blue-500/20 border-blue-500 text-blue-600' : 'bg-red-500/20 border-red-500 text-red-500';

                                return (
                                    <div key={block.id}
                                         className={`absolute bottom-10 flex flex-col border-t-2 border-r-2 ${colorClass} transition-colors group`}
                                         style={{
                                             left: `${leftPercent}%`,
                                             width: `${widthPercent}%`,
                                             height: `${heightPercent}%`, // relative to the area
                                             borderLeftWidth: '1px',
                                             borderLeftColor: 'rgba(255,255,255,0.1)'
                                         }}>

                                        {/* Block Label / Header drag handle */}
                                        <div
                                            className="w-full p-2 bg-white/80 dark:bg-slate-800/80 backdrop-blur-md flex justify-between items-start cursor-ns-resize select-none"
                                            onPointerDown={(e) => handlePointerDown(e, block, 'height')}
                                        >
                                            <input
                                                type="text"
                                                value={block.name}
                                                onChange={(e) => {
                                                    const newBlocks = [...blocks];
                                                    newBlocks[idx].name = e.target.value;
                                                    setBlocks(newBlocks);
                                                }}
                                                className="text-xs font-bold bg-transparent outline-none w-full pointer-events-auto"
                                                onPointerDown={(e) => e.stopPropagation()} // Let typing work
                                            />
                                            {idx > 0 && (
                                                <button
                                                    onPointerDown={(e) => e.stopPropagation()}
                                                    onClick={() => setBlocks(blocks.filter(b => b.id !== block.id))}
                                                    className="opacity-0 group-hover:opacity-100 text-slate-400 hover:text-red-500 p-1 pointer-events-auto"
                                                >
                                                    <Icon name="Trash2" size={12} />
                                                </button>
                                            )}
                                        </div>

                                        {/* Right Edge Drag Handle for Duration */}
                                        <div
                                            className="absolute right-0 top-0 bottom-0 w-2 cursor-ew-resize hover:bg-black/10 dark:hover:bg-white/10"
                                            onPointerDown={(e) => handlePointerDown(e, block, 'duration')}
                                        ></div>

                                    </div>
                                );
                            })}

                            {/* SVG Layer for Progression Line */}
                            {showProgression && (
                                <svg className="absolute inset-0 w-full h-full pointer-events-none z-20" style={{ paddingBottom: '2.5rem', paddingTop: '2rem' }} preserveAspectRatio="none">
                                    {blocks.map(block => {
                                        // Draw lines between weeks
                                        if (block.weeklyData.length < 2) return null;

                                        let pathD = "";
                                        const weekWidth = 100 / totalTimelineWeeks;

                                        block.weeklyData.forEach((w, i) => {
                                            const x = ((w.week - 1 + 0.5) / totalTimelineWeeks) * 100;
                                            const val = timelineView === 'Volumen' ? w.volume : w.intensity;
                                            const y = 100 - val; // Invert Y for SVG

                                            if (i === 0) pathD += `M ${x} ${y} `;
                                            else pathD += `L ${x} ${y} `;
                                        });

                                        return (
                                            <g key={`svg-${block.id}`}>
                                                <path d={pathD} fill="none" stroke={timelineView === 'Volumen' ? "#3b82f6" : "#ef4444"} strokeWidth="2" vectorEffect="non-scaling-stroke" />
                                                {/* Dots for each week */}
                                                {block.weeklyData.map((w, i) => {
                                                    const x = ((w.week - 1 + 0.5) / totalTimelineWeeks) * 100;
                                                    const val = timelineView === 'Volumen' ? w.volume : w.intensity;
                                                    const y = 100 - val;
                                                    return (
                                                        <circle
                                                            key={`dot-${w.week}`}
                                                            cx={`${x}%`}
                                                            cy={`${y}%`}
                                                            r="6"
                                                            fill="white"
                                                            stroke={timelineView === 'Volumen' ? "#3b82f6" : "#ef4444"}
                                                            strokeWidth="2"
                                                            className="pointer-events-auto cursor-ns-resize hover:r-8 hover:stroke-4 transition-all"
                                                            onPointerDown={(e) => handlePointerDown(e, block, 'weeklyPoint', i)}
                                                        />
                                                    );
                                                })}
                                            </g>
                                        );
                                    })}
                                </svg>
                            )}

                            {/* Append Block Button (Plus on Timeline) */}
                            <div className="absolute bottom-10 flex items-center justify-center z-10"
                                 style={{ left: `${(durationWeeks / totalTimelineWeeks) * 100}%`, marginLeft: '8px' }}>
                                <button
                                    onClick={handleAddBlock}
                                    className="w-8 h-8 rounded-full bg-white dark:bg-slate-800 shadow-md border border-slate-200 dark:border-slate-700 text-slate-400 hover:text-indigo-600 hover:border-indigo-500 flex items-center justify-center transition-colors"
                                    title="Neuen Block anfügen"
                                >
                                    <Icon name="Plus" size={16} />
                                </button>
                            </div>

                        </div>

                        {/* X-Axis (Weeks Grid) */}
                        <div className="absolute bottom-0 left-0 right-0 h-10 border-t border-slate-200 dark:border-slate-700 bg-white dark:bg-slate-800 flex items-center z-20">
                            {Array.from({ length: totalTimelineWeeks }).map((_, i) => (
                                <div key={i} className="flex-1 flex justify-center px-1 border-r border-slate-100 dark:border-slate-700/50 last:border-r-0 h-full items-center">
                                    <div className="text-[10px] font-bold text-slate-400">W{i+1}</div>
                                </div>
                            ))}
                        </div>
                    </div>
                 </div>



                 {/* Linked Plans Area - Sub-timelines */}
                 <div className="mt-4 shrink-0 bg-slate-50 dark:bg-slate-900/50 border border-slate-200 dark:border-slate-700 rounded-xl overflow-hidden relative transition-all duration-300">
                    <div
                        className="text-xs font-bold text-slate-500 dark:text-slate-400 p-4 flex items-center justify-between cursor-pointer hover:bg-slate-100 dark:hover:bg-slate-800"
                        onClick={() => setIsLinkedPlansExpanded(!isLinkedPlansExpanded)}
                    >
                        <span className="flex items-center gap-2"><Icon name="List" size={14}/> Verlinkte Pläne bearbeiten</span>
                        <Icon name={isLinkedPlansExpanded ? "ChevronUp" : "ChevronDown"} size={16} />
                    </div>

                    {isLinkedPlansExpanded && (
                    <div className="relative w-full pb-4 px-4 flex flex-col gap-4">
                        <div className="text-sm text-slate-500 dark:text-slate-400">
                            Füge Pläne hinzu, die dann in der Timeline oben als Balken erscheinen.
                        </div>
                        <div className="flex gap-4">
                            <button
                                onClick={() => setTrainingPlans([...trainingPlans, { id: 'tp-'+Date.now(), name: 'Neuer TP', startWeek: 1, durationWeeks: 4 }])}
                                className="px-4 py-2 bg-blue-500 hover:bg-blue-600 text-white font-bold rounded-lg flex items-center gap-2 text-sm shadow-sm"
                            >
                                <Icon name="Dumbbell" size={16} />
                                Trainingsplan hinzufügen
                            </button>
                            <button
                                onClick={() => setNutritionPlans([...nutritionPlans, { id: 'ep-'+Date.now(), name: 'Neuer EP', startWeek: 1, durationWeeks: 4 }])}
                                className="px-4 py-2 bg-emerald-500 hover:bg-emerald-600 text-white font-bold rounded-lg flex items-center gap-2 text-sm shadow-sm"
                            >
                                <Icon name="Utensils" size={16} />
                                Ernährungsplan hinzufügen
                            </button>
                        </div>
                    </div>
                    )}
                 </div>
            </div>
        </div>
    );

        function App() {
            const [activeTab, setActiveTabState] = useState('Dashboard');
            const [isDarkMode, setIsDarkMode] = useState(true);
            const [isSidebarOpen, setIsSidebarOpen] = useState(true);
            const [athletesData, setAthletesData] = useState([]);
            const [isLoadingAthletes, setIsLoadingAthletes] = useState(false);
            const [selectedAthlete, setSelectedAthlete] = useState(null);

            // Custom setActiveTab to handle passing data
            const setActiveTab = (tabName, data = null) => {
                setActiveTabState(tabName);
                if (data) setSelectedAthlete(data);
            };

            useEffect(() => {
                if (isDarkMode) {
                    document.documentElement.classList.add('dark');
                } else {
                    document.documentElement.classList.remove('dark');
                }
            }, [isDarkMode]);

            // Fetch Athletes on mount
            useEffect(() => {
                setIsLoadingAthletes(true);
                if (typeof google !== 'undefined' && google.script && google.script.run) {
                    google.script.run
                        .withSuccessHandler((res) => {
                            if (res.success) {
                                setAthletesData(res.data);
                            } else {
                                console.error("Failed to load athletes:", res.error);
                            }
                            setIsLoadingAthletes(false);
                        })
                        .withFailureHandler((err) => {
                            console.error("Error calling fetchAllAthletes:", err);
                            setIsLoadingAthletes(false);
                        })
                        .fetchAllAthletes();
                } else {
                    // Local mock fallback
                    setTimeout(() => {
                        setAthletesData([
                            { id: 'ATH_DENIZ_001', name: 'Deniz', status: 'Active', nextCheckIn: 'Montag', tags: ['Pro'] },
                            { id: 'ATH_MERET_002', name: 'Meret', status: 'Active', nextCheckIn: 'Mittwoch', tags: ['Basic'] }
                        ]);
                        setIsLoadingAthletes(false);
                    }, 500);
                }
            }, []);


            return (
                <div className="min-h-screen bg-gray-50 dark:bg-slate-950 text-slate-800 dark:text-slate-100 flex transition-colors duration-300 font-sans">
                    <Sidebar activeTab={activeTab} setActiveTab={setActiveTab} isOpen={isSidebarOpen} setIsOpen={setIsSidebarOpen} />

                    {/* Main Content Area */}
                    <div className={`flex-1 flex flex-col transition-all duration-300 ${isSidebarOpen ? 'ml-64' : 'ml-20'}`}>
                        <Header isDarkMode={isDarkMode} setIsDarkMode={setIsDarkMode} isSidebarOpen={isSidebarOpen} setIsSidebarOpen={setIsSidebarOpen} />

                        <main className="w-full relative z-0 p-6">

                            {activeTab === 'Dashboard' && <Dashboard data={mockData} />}
                            {activeTab === 'Athletes' && <Athletes setActiveTab={setActiveTab} athletesData={athletesData} isLoadingAthletes={isLoadingAthletes} />}
                            {activeTab === 'AthleteDetail' && <AthleteDetail setActiveTab={setActiveTab} athlete={selectedAthlete} />}
                            {activeTab === 'Planner' && <Planner setActiveTab={setActiveTab} selectedAthlete={selectedAthlete} />}

                            {activeTab === 'CheckIns' && <CheckIn />}
                            {activeTab === 'Calls' && <Calls />}
                            {activeTab !== 'Dashboard' && activeTab !== 'Athletes' && activeTab !== 'AthleteDetail' && activeTab !== 'Planner' && activeTab !== 'CheckIns' && activeTab !== 'Calls' && (
                                <div className="flex items-center justify-center h-64 text-slate-500 dark:text-slate-400">
                                    <p className="text-lg font-semibold">Ansicht für {activeTab} wird geladen...</p>
                                </div>
                            )}
                        </main>
                    </div>
                    <LiveTicketSystem />
                </div>
            );
        }
